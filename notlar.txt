func GetUserByWalletAddress(db *gorm.DB, walletAddress string) (*model.User, error) {
	var user model.User
	if err := db.Where("wallets.wallet_address = ?", walletAddress).
		Joins("JOIN wallets ON wallets.user_id = users.id").
		First(&user).Error; err != nil {
		return nil, err
	}
	return &user, nil
}
